# =====================================================================
# SamQL backend dependencies
# =====================================================================
# The SamQL backend runs on the Python standard library alone. None of
# the packages below are required to load CSV/JSON files, run SQL on the
# built-in SQLite engine, profile tables, chart results, or serve the UI.
#
# Install only what you need. Each unlocks an optional capability and is
# detected at runtime (and bundled by PyInstaller only if present).
#
#   pip install -r requirements.txt
#
# ---------------------------------------------------------------------
# Optional acceleration / format support
# ---------------------------------------------------------------------
# duckdb         # fast analytical engine + native Parquet querying
# pyarrow        # Parquet read/write, Arrow interchange
# sqlglot        # SQL formatting and cross-dialect transpilation
# openpyxl       # .xlsx export
# orjson         # faster JSON: API/JSON load parsing + result-page encoding
# ijson          # streaming parser for very large JSON arrays (lower memory)
# pyodbc         # connect to Microsoft SQL Server
# pywebview      # open SamQL in a native desktop window instead of a tab
#
# Windows-only, for SQL Server integrated/Windows auth + impersonation:
# pywin32
#
# ---------------------------------------------------------------------
# Build-time only (not needed to run from source)
# ---------------------------------------------------------------------
# pyinstaller    # produce a single distributable executable
