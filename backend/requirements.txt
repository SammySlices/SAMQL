# =====================================================================
# SamQL backend dependencies

# Core runtime extras that every source install should have. Distribution
# builds also install requirements-optional.txt (DuckDB, pyarrow, …) and
# hard-verify those imports before packaging.
#
#   pip install -r backend/requirements.txt
#   pip install -r requirements-optional.txt   # full analytics stack
#
# orjson is required for fast JSON array → NDJSON rewrite and API payloads.
orjson>=3.9.0

# ---------------------------------------------------------------------
# Optional acceleration / format support (see requirements-optional.txt)
# ---------------------------------------------------------------------
# duckdb         # fast analytical engine + native Parquet querying
# pyarrow        # Parquet read/write, Arrow interchange
# pandas         # Python node DataFrames (bundled in distribution builds)
# sqlglot        # SQL formatting and cross-dialect transpilation
# openpyxl       # .xlsx export
# ijson          # streaming parser for very large JSON arrays (lower memory)
# tzdata         # IANA tz database (zoneinfo on Windows / frozen builds)
# pytz           # DuckDB TIMESTAMPTZ → Python (required by DuckDB bindings)
# pyodbc         # connect to Microsoft SQL Server
# pywebview      # open SamQL in a native desktop window instead of a tab
#
# Windows-only, for SQL Server integrated/Windows auth + impersonation:
# pywin32
